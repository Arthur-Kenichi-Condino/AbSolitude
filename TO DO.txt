class MonoPool<T> where T : MonoBehaviour
{
    private readonly Stack<T> pool = new();
    private readonly T prefab;
    private readonly Transform parent;

    public MonoPool(T prefab, Transform parent = null)
    {
        this.prefab = prefab;
        this.parent = parent;
    }

    public T Get()
    {
        if (pool.Count > 0)
        {
            var item = pool.Pop();
            item.gameObject.SetActive(true);
            return item;
        }

        return GameObject.Instantiate(prefab, parent);
    }

    public void Release(T item)
    {
        item.gameObject.SetActive(false);
        pool.Push(item);
    }
}

CalculateChunkSet(HashSet<int> output)
    output.Clear();

    Vector3 pos = transform.position;

    int centerX = WorldToChunkX(pos.x);
    int centerY = WorldToChunkY(pos.z);

    for(int y = -instY; y <= instY; y++)
    {
        int cy = centerY + y;
        if(Mathf.Abs(cy) >= MaxcCoordy)
            continue;

        for(int x = -instX; x <= instX; x++)
        {
            int cx = centerX + x;
            if(Mathf.Abs(cx) >= MaxcCoordx)
                continue;

            int chunkId = GetChunkId(cx, cy);
            output.Add(chunkId);
        }
    }
